```bash

#############################################
# PART 1: COMMIT PHASE
#############################################
echo ""

    echo "Nothing to commit. Checking for unpushed commits..."

    if [ ! -f .gitignore ]; then

        [[ ! $REPLY =~ ^[Yy]$ ]] && exit 1

    echo "üîç Scanning for sensitive patterns..."
    if git diff | grep -qiE "(api[_-]?key|password|token|secret|private[_-]?key)"; then

        git diff | grep -iE "(api[_-]?key|password|token|secret)" | head -5

        echo
        [[ ! $REPLY =~ ^[Yy]$ ]] && exit 1
    fi

    # 4. Stage changes
    echo ""
    echo "üì¶ Files with changes:"
    git status --short
    echo ""
    echo "How would you like to stage changes?"
    echo "  [a] Stage all changes"
    echo "  [i] Interactive staging"
    echo "  [s] Select specific files"
    echo "  [q] Quit"

            echo "‚úì All changes staged"
            ;;
        i|I)

            ;;

            ;;

    echo "üìù Staged changes:"

    echo "  ‚Ä¢ Capitalize the subject line"
    echo "  ‚Ä¢ No period at the end"
    echo "  ‚Ä¢ Explain what and why, not how"
    echo ""
    echo "Ready to write commit message..."
    sleep 1
    
    # Commit
    git commit -v
    
    if [ $? -ne 0 ]; then

    echo -e "${GREEN}‚úì Commit created successfully${NC}"

#############################################

#############################################
echo ""

            ;;

            ;;

            exit 0
            ;;
    esac
```
